💻 EXEMPLO COMPLETO PRONTO PARA USAR
═══════════════════════════════════════════════════════════════════════════════

Copie e adapte para seus componentes:

import { useState } from "react";
import { budgetActions, useLoading } from "@/lib/budget-store";
import { LoadingButton, Spinner } from "@/components/ui/loading";
import { crudToasts } from "@/lib/toast-helper";

export function CreateStageForm() {
  const [name, setName] = useState("");
  const [planned, setPlanned] = useState("");
  const loading = useLoading();

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    
    if (!name || !planned) {
      toastError("Preencha todos os campos");
      return;
    }

    try {
      await crudToasts.create(
        budgetActions.addStage(name, parseFloat(planned)),
        "Etapa"
      );
      setName("");
      setPlanned("");
    } catch (error) {
      console.error(error);
    }
  };

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <input
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Nome da etapa"
        disabled={loading.creating}
        className="w-full px-3 py-2 border rounded"
      />
      
      <input
        type="number"
        value={planned}
        onChange={(e) => setPlanned(e.target.value)}
        placeholder="Valor planejado"
        disabled={loading.creating}
        className="w-full px-3 py-2 border rounded"
      />

      <LoadingButton
        type="submit"
        isLoading={loading.creating}
        loadingText="Criando etapa..."
        className="w-full"
      >
        Criar Etapa
      </LoadingButton>
    </form>
  );
}

💡 DICAS
═══════════════════════════════════════════════════════════════════════════════

1. Use crudToasts.create/update/delete para a maioria dos casos
   → Automático, consistente, sem código extra

2. Use LoadingButton sempre que tiver operações assíncronas
   → Previne double-clicks, mostra feedback visual

3. Use Spinner para listas/tabelas carregando
   → Melhor que placeholder

4. Use LoadingOverlay para diálogos/modais
   → Fica mais visual, não deixa fazer mudanças enquanto processa

5. Se precisar toast manual:
   toastLoading() → fazer operação → toastUpdateLoading()
